inline void solve1()
{
    ll n;
    cin>>n;
    ll arr[n];
    for(ll i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    sort(arr,arr+n);
    ll ans=-INF;

     for(ll i=0;i<n;i++)
     {
         ll a,b,c,d,e;
         a=(n-1+i)%n;
         b=(n-2+i)%n;
         c=(n-3+i)%n;
         d=(n-4+i)%n;
         e=(n-5+i)%n;
         ll v=arr[a]*arr[b]*arr[c]*arr[d]*arr[e];
         ans=max(ans,v);
     }
     cout<<ans<<endl;

}
