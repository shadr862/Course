#include <bits/stdc++.h>
using namespace std;
int main() {
    long long n,i,j;
    cin>>n;
    long long arr[n+1];
    long long arr1[n+1][n+1];
    for(i=1;i<=n;i++)
    {
        cin>>arr[i];
        arr1[i][i]=0;
    }
    sort(arr+1,arr+n+1);
    for(i=2;i<=n;i++)
    {
        for(j=1;j<=n-i+1;j++)
        {
            arr1[j][j+i-1]=(arr[j+i-1]-arr[j])+min(arr1[j+1][j+i-1],arr1[j][j+i-2]);
        }
    }
    cout<<arr1[1][n]<<endl;
}
