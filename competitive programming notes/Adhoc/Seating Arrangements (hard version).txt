ll res(vector<pair<ll,ll> > v,ll m)
{
    sort(all(v),sorta);
    ll c=0;
    for(ll i=1;i<m;i++)
    {

        for(ll j=0;j<i;j++)
        {
            if(v[i].ff>v[j].ff)
            {
                c++;
            }

        }

    }
    return c;
}
void solve()
{

    for(ll i=0;i<n*m;i++)

        v[i]={x,i};
    }
    sort(all(v));
    ll ii=0;
    ll jj=m;
    ll r=0;
    for(ll i=0;i<n;i++)
    {
        vector<pair<ll,ll> > v1(v.begin()+ii,v.begin()+jj);
        ii+=m;
        jj+=m;
        r+=res(v1,m);

    }

}
