ll arr[N];
ll tree[N * 3];
void build(ll node,ll b,ll e)
{
    if(b==e)
    {
        tree[node]=arr[b];
        return ;
    }
    ll left=node*2;
    ll right=node*2+1;
    ll mid=(b+e)/2;
    build(left,b,mid);
    build(right,mid+1,e);
    tree[node]=__gcd(tree[left],tree[right]);
}


ll query(int  node, int b, int e, int i, int j)
{
    if(i>e||j<b)
    {
        return 0;
    }

    if(b>=i && e<=j)
    {
        return tree[node];
    }
    ll left=node*2;
    ll right=node*2+1;
    ll mid=(b+e)/2;
    ll p1=query(left,b,mid,i,j);
    ll p2=query(right,mid+1,e,i,j);

    return __gcd(p1,p2);
}
