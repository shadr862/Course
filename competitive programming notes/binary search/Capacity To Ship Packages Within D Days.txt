bool ok(ll mid , ll weights[] ,ll n, ll D){

        ll days=1;
        ll total=0;
        bool isValid=true;
        for(int i=0;i<n;i++){
            if(weights[i]>mid)
                isValid=false;
            if(total+weights[i]<=mid)
                total+=weights[i];
            else{
                days++;
                total=weights[i];
            }
        }
        if(!isValid)
            return false;
        else
            return (days<=D);
    }

}
