{

}




inline int solve1()
{
    ll n;
    cin>>n;

        ll l=1;
        ll r=1<<31-1;
               while(l<r){
               ll mid=(l+r)/2;
               ll ans=(ll)mid*mid;
               if(ans==(ll)n)
                  return 1;
               else if(ans>(ll)n)
                   r=mid;
               else
                   l=mid+1;
          }

          if((ll)l*l==(ll)n)return 1;

           return 0;

}
