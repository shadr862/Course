typedef long long int ll;
typedef unsigned long long int  llu;
#define mod 1000000007
ll fibonacci(ll a,ll b,ll n)
{
	ll mat[n+1];
	ll i;
	mat[0]=a%mod;
	mat[1]=b%mod;
	for(i=2;i<=n;i++)
	{
		mat[i]=mat[i-1]%mod+mat[i-2]%mod;
	}
	return mat[n]%mod;
}

		cout<<fibonacci(a,b,n)<<nl;
	}
