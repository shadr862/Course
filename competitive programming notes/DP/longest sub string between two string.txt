#include <bits/stdc++.h>
using namespace std;
int main() {
    int t;
    cin>>t;
    while(t--)
    {
        string s1,s2;
        cin>>s1>>s2;
        int r=s1.size();
        int c=s2.size();
        int dp[r+1][c+1];
        int result=0,r2=0;
        if(s1==s2)
        {
            cout<<0<<endl;
            continue;
        }
        for(int i=0;i<=r;i++)
        {
            for(int j=0;j<=c;j++)
            {
            	if(i==0||j==0)
            	{
            		dp[i][j]=0;
            	}
                 else if(s1[i-1]==s2[j-1])
                {
                    dp[i][j]=dp[i-1][j-1]+1;
                    result=max(result,dp[i][j]);
                }
                else{
                    dp[i][j]=0;
                }
 
            }
        }
        
        {
            r2=r-result+c-result;
        }
        
        cout<<r2<<endl;
    }
}
