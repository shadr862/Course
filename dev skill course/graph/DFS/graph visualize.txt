#include <bits/stdc++.h>
#define LIM 1003

using namespace std;

/**
n = number of people
m = number of relations
*/
int n, m, friends[LIM][LIM];
vector<int> g[LIM];
int vis[LIM];

void mat_input() {
    scanf("%d %d", &n, &m);
    for(int i=0; i<m; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        friends[u][v] = friends[v][u] = 1;
    }
}

void list_input() {
    scanf("%d %d", &n, &m);
    for(int i=0; i<m; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
}

void dfs(int u) {
    cout << "in " << u << endl;
    getchar();
    vis[u] = 1;
    for(int i=0; i<g[u].size(); i++) {
        int v = g[u][i];
         cout << "check " << v << ' ' << vis[v] << endl;
         getchar();
         if(vis[v]) continue;
         cout << "going from " << u << " to " << v << endl;
         getchar();
         dfs(v);
    }
        cout << "out " << u << endl;
        getchar();
}

    list_input();
    memset(vis, 0, sizeof vis);
    for(int u=1; u<=n; u++) {
        if(!vis[u]) {
            
        }
    }
}
