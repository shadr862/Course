#include <bits/stdc++.h>
#define LIM 1003
using namespace std;

int n,m,c=1;
vector<int> g[LIM],weight[LIM];
int vis[LIM],dis[LIM];

void list_input() {
    scanf("%d", &n);
    for(int i=0; i<n; i++){
        g[i].clear();
        weight[i].clear();
    }
    m=n-1;
    for(int i=0; i<m; i++) {
        int u, v, c;
        scanf("%d %d %d", &u, &v,&c);
        g[u].push_back(v);
        g[v].push_back(u);
        weight[u].push_back(c);
        weight[v].push_back(c);
    }
}
void clr()
{
    for( int i = 0 ; i < n ; i++ )
    {
        vis[i] = 0;
        dis[i] = 0;
    }
}

void bfs(int root) {
    queue<int> q;
    vis[root]=1;
    q.push(root);

    while(!q.empty()) {
        int u = q.front();
        q.pop();
        for(int i=0; i<g[u].size(); i++) {
            int v = g[u][i];
            if(vis[v])
            {
               continue;
            }

            vis[v]=1;
            dis[v]=dis[u]+weight[u][i];
            q.push(v);
        }
    }

}


void solve()
{
    list_input();
    bfs(0);

    int f_node,mx=-1;

    for(int i=0;i<n;i++)
    {
        if(mx<dis[i])
        {
            mx=dis[i];
            f_node=i;
        }
    }

    clr();
    mx=0;
    bfs(f_node);

    for(int i=0;i<n;i++)
    {
        if(mx<dis[i])
        {
            mx=dis[i];
            f_node=i;
        }
    }

   cout<<"Case "<<c<<": "<<mx<<endl;
    clr();
}



int main() {
  int t;
  cin>>t;
  while(t--)
  {
      solve();
      c++;
  }

}
