#include <bits/stdc++.h>
#define LIM 1003
using namespace std;

int n,m;
vector<int> g[LIM];
int vis[LIM], dis[LIM];

void list_input() {
    memset(vis, 0, sizeof vis);
    memset(par, 0, sizeof par);

    scanf("%d %d", &n, &m);
    for(int i=1; i<=n; i++) g[i].clear();

    for(int i=0; i<m; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
}

void bfs(int root) {
    vis[root] = 1;
    dis[root] = 0;
    queue<int> q;
    q.push(root);

    while(!q.empty()) {
        int u = q.front();
        q.pop();
        for(int i=0; i<g[u].size(); i++) {
            int v = g[u][i];
            if(vis[v]) continue;
            vis[v] = 1;
            dis[v] = dis[u]+1;
            q.push(v);
        }
    }
}

void print_dis() {
    for(int u=1; u<=n; u++)
        printf("%d : %d\n", u, dis[u]);
}

int main() {
    list_input();
    for(int u=1; u<=n; u++) {
        if(!vis[u]) {
            bfs(u);
        }
    }
    print_dis();
}
