#include <bits/stdc++.h>
#define LIM 1003
using namespace std;

int n,m;
vector<int> g[LIM];
int vis[LIM],par[LIM];

void list_input() {
    memset(vis, 0, sizeof vis);
    memset(par,0,sizeof par);
    scanf("%d %d", &n, &m);
    for(int i=1; i<=n; i++) g[i].clear();

    for(int i=0; i<m; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
}

bool bfs(int root,int p) {
    queue<int> q;
    vis[root]=1;
    q.push(root);
    par[root]=p;

    while(!q.empty()) {
        int u = q.front();
        q.pop();
        for(int i=0; i<g[u].size(); i++) {
            int v = g[u][i];
            par[v]=u;
            if(v==par[u])continue;
            if(vis[v])
            {
                cout<<par[u]<<" "<<par[v]<<endl;
                cout<<u<<" "<<v<<endl;
                return 1;
            }
            vis[v]=1;
            q.push(v);
        }
    }
    return false;
}




int main() {
    list_input();
    int f=1;
    for(int u=1; u<=n; u++) {
        if(!vis[u]) {
            if(bfs(u,0)==1){
                cout<<"Cycle detected"<<endl;
                f=0;
                break;
            }
        }
    }
    if(f==1)cout<<"Cycle not detected"<<endl;

}
