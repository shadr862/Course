void solve()
{

        m[s[i]]++;
    }
    ll c=0;
    for(auto x:m)
    {
        if(x.second%2==1)c++;
    }

    if(c>1)
    {
        cout<<"NOt possible to make palindrom"<<endl;
    }

    {
        cout<<"possible to make palindrom"<<endl;
    }

}
