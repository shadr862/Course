#include <bits/stdc++.h>
#define LIM 1003

using namespace std;

int n, cap, coin[LIM];
int dp[LIM][LIM],dr[LIM][LIM];

int f(int in, int rem) {

    if(rem==0)
        return 1;
    if(in >= n)
        return 0;

    if(dp[in][rem] ==1)
        return dp[in][rem];


    int ans = f(in+1, rem);
    if(coin[in] <= rem)
    {
        int temp=f(in+1,rem-coin[in]);
        ans|=temp;
    }
    return dp[in][rem] = ans;
}

int main() {
    scanf("%d %d", &n, &cap);
    for(int i=0; i<n; i++)
        scanf("%d", &coin[i]);

    memset(dp, -1, sizeof dp);


    for(int i=1;i<=17;i++)
    {
        if(f(0,i))
        {
           cout<<i<<":possible"<<endl;
        }
        else
        {
           cout<<i<<":not possible"<<endl;
        }
    }
}
